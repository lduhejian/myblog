c 语言代码判断机器是大端还是小端：
```
#include <stdio.h>

char* endian()
{
        int v = 0x0100;
        if ((char)v == 1)
        {
                return "Big Endian";
        }
        else {
                return "Little Endian";
        }
}

char* endian_by_union()
{
        union u {
                int a;
                char b;
        } u1;
        u1.a = 0x0;
        u1.a = 0x0001;
        if (u1.b == 1) {
                return "Little Endian";
        }
        else {
                return "Big Endian";
        }
}

int main()
{

        printf("endian(): %s.\n", endian());
        printf("endian_by_union(): %s.\n", endian_by_union());

        return 0;
}
```
